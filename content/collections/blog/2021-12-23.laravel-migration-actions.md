          use Illuminate\Support\Facades\DB;
           
          class ExampleAction extends Actionable
          {
              protected $transactions = true;
           
              /**
               * Run the actions.
               *
               * @return void
               */

              {
                  DB::table('users')->insert([
                      'name' => 'Example User',
                      'email' => 'user@example.com',
                      'password' => bcrypt('password')
                  ]);
              }

               * Reverse the actions.

          }
